                    ${CMAKE_CURRENT_SOURCE_DIR}
                    ${CMAKE_CURRENT_LIST_DIR}/..
                    ${CMAKE_BINARY_DIR}
                    ${CMAKE_CURRENT_BINARY_DIR}/..)

    kcm.cpp
)

    KF5::I18n
    KF5::KCMUtils
    kdeconnectinterfaces
